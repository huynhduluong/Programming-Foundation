var img = new SimpleImage("smallpanda.png");
//setblack
function setBlack(pixel){
    pixel.setRed(0);
    pixel.setGreen(0);
    pixel.setBlue(0);
    return pixel;
}
for(var pixel of img.values()){
    var x = pixel.getX();
    var y = pixel.getY();
    
    if(x <= 10 || x >= img.getWidth()-10){
        setBlack(pixel);
    }
    
    if(y <= 10 || y >= img.getHeight()-10){
        setBlack(pixel);
    }
}

print(img);
